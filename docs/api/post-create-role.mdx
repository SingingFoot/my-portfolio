export const HighlightedCode = ({ code }) => {
  const highlight = (text) => {
    let highlighted = text
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/(\/\/.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/(#.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/("(?:[^"\\]|\\.)*")|('(?:[^'\\]|\\.)*')/g, '<span style="color: #a5d6ff">$1</span>')
      .replace(/\b(import|from|const|await|new|var|require|return|if|else|public|private|class|interface|package|use|namespace|def|func|true|false|null)\b/g, '<span style="color: #ff7b72">$1</span>')
      .replace(/\b(TechWriter|TechWriterClient|RequestOptions|Role|RoleParams|context|ctx|options|client|tw|sc|service)\b/g, '<span style="color: #ffa657">$1</span>')
      .replace(/\b([a-z_][a-z0-9_]*)(?=\s*\()/gi, '<span style="color: #d2a8ff">$1</span>')
      .replace(/\b(\d+)\b/g, '<span style="color: #79c0ff">$1</span>');

    return <code dangerouslySetInnerHTML={{ __html: highlighted }} />;
  };

  return (
    <pre style={{ margin: 0, background: 'transparent', whiteSpace: 'pre-wrap', fontFamily: 'monospace' }}>
      {highlight(code)}
    </pre>
  );
};

              <div style={{ padding: '2rem', textAlign: 'center', fontSize: '13px', color: '#88888f' }}>No privileges match your filter.</div>
            ) : (
              filteredPrivileges.map((p, idx) => (
                <div key={p.id} style={{ display: 'grid', gridTemplateColumns: '140px 1fr 60px 1.4fr', gap: '1rem', padding: '0.75rem 1rem', borderBottom: idx === filteredPrivileges.length - 1 ? 'none' : '1px solid #1e1e20', alignItems: 'baseline', fontSize: '13px' }}>
                  <span style={{ color: '#88888f', fontSize: '12px' }}>{p.cat}</span>
                  <span style={{ color: '#e3e3e8' }}>{p.name}</span>
                  <code style={{ textAlign: 'right', fontFamily: 'monospace', color: '#F68B1F', fontWeight: '700' }}>{p.id}</code>
                  <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{p.desc}</span>
                </div>
              ))
            )}
          </div>

          {/* Response Codes */}
          <h2 style={{ fontSize: '20px', fontWeight: 'bold', color: 'var(--ifm-font-color-base)', marginBottom: '0.75rem', letterSpacing: '-0.02em' }}>Response codes</h2>
          <p style={{ fontSize: '14px', color: '#adadb8', marginBottom: '1rem' }}>The API uses conventional HTTP status codes. <code style={{ backgroundColor: '#1e1e20', padding: '0.125rem 0.375rem', borderRadius: '0.25rem', fontFamily: 'monospace', color: '#e3e3e8', fontSize: '13px' }}>2xx</code> indicate success, <code style={{ backgroundColor: '#1e1e20', padding: '0.125rem 0.375rem', borderRadius: '0.25rem', fontFamily: 'monospace', color: '#e3e3e8', fontSize: '13px' }}>4xx</code> indicate client errors, and <code style={{ backgroundColor: '#1e1e20', padding: '0.125rem 0.375rem', borderRadius: '0.25rem', fontFamily: 'monospace', color: '#e3e3e8', fontSize: '13px' }}>5xx</code> indicate server-side issues.</p>
          <div style={{ marginBottom: '3rem', borderTop: '1px solid #1e1e20' }}>
            {statusOrder.map((s) => (
              <div key={s} style={{ display: 'grid', gridTemplateColumns: '160px 1fr', gap: '1rem', padding: '0.875rem 0', borderBottom: '1px solid #1e1e20', alignItems: 'baseline' }}>
                <span style={{ fontFamily: 'monospace', color: statusAccent(s), fontSize: '13px', fontWeight: '700' }}>{responseExamples[s].label}</span>
                <span style={{ fontSize: '13px', color: '#adadb8', lineHeight: '1.5' }}>{responseExamples[s].desc}</span>
              </div>
            ))}
          </div>

          {/* Feedback Section */}
          <div style={{ marginTop: '4rem', paddingTop: '3rem', borderTop: '1px solid #1e1e20' }}>
            {!feedback ? (
              <div style={{ display: 'flex', alignItems: 'center', gap: '2rem' }}>
                <span style={{ fontSize: '14px', fontWeight: '500', color: '#88888f' }}>Was this section helpful?</span>
                <button onClick={() => setFeedback('yes')} style={{ color: '#F68B1F', fontSize: '14px', fontWeight: '600', background: 'none', border: 'none', cursor: 'pointer' }}>Yes</button>
                <button onClick={() => setFeedback('no')} style={{ color: '#F68B1F', fontSize: '14px', fontWeight: '600', background: 'none', border: 'none', cursor: 'pointer' }}>No</button>
              </div>
            ) : (
              <div style={{ animation: 'fadeIn 0.5s' }}>
                {feedback === 'yes' ? (
                  <p style={{ fontSize: '14px', color: '#e3e3e8', backgroundColor: '#1a1a1c', padding: '0.75rem 1.5rem', borderRadius: '0.375rem', display: 'inline-block', border: '1px solid #303033' }}>
                    ✨ Thanks for your feedback! We're glad this helped.
                  </p>
                ) : (
                  <div style={{ backgroundColor: '#1a1a1c', padding: '1.5rem', borderRadius: '0.375rem', border: '1px solid #303033', maxWidth: '448px' }}>
                    <p style={{ fontSize: '14px', color: '#e3e3e8', marginBottom: '1rem' }}>We're sorry! How can we improve this page?</p>
                    <textarea
                      style={{ width: '100%', backgroundColor: '#0a0a0a', border: '1px solid #303033', borderRadius: '0.5rem', padding: '0.75rem', fontSize: '0.875rem', color: '#adadb8', outline: 'none' }}
                      placeholder="Your feedback..."
                      rows="4"
                    ></textarea>
                    <button style={{ marginTop: '1rem', backgroundColor: '#F68B1F', color: 'white', fontSize: '0.75rem', fontWeight: 'bold', padding: '0.5rem 1rem', borderRadius: '0.5rem', border: 'none', cursor: 'pointer' }}>Submit Feedback</button>
                  </div>
                )}
              </div>
            )}
            <div style={{ paddingBottom: '8rem' }}></div>
          </div>
        </div>
      </div>

      {/* Right Column: Code and Responses */}
      <div style={{ width: isMobileView ? '100%' : '45%', backgroundColor: 'rgba(0,0,0,0.02)', padding: isMobileView ? '1.5rem 1rem' : '3rem', borderLeft: isMobileView ? 'none' : '1px solid #1e1e20', borderTop: isMobileView ? '1px solid #1e1e20' : 'none' }}>
        <div style={{ position: isMobileView ? 'static' : 'sticky', top: isMobileView ? 'auto' : '3rem', display: 'flex', flexDirection: 'column', gap: '2rem' }}>

          {/* Language Picker */}
          <div style={{ position: 'relative' }}>
            <button
              onClick={() => setIsOpen(!isOpen)}
              style={{
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'space-between',
                width: '100%',
                backgroundColor: '#1a1a1c',
                padding: '0.75rem 1.25rem',
                borderRadius: '0.5rem',
                fontSize: '14px',
                color: 'white',
                border: '1px solid #303033',
                cursor: 'pointer'
              }}
            >
              <span style={{ fontWeight: '600' }}>{language}</span>
              <ChevronDown size={16} style={{ color: '#88888f', transform: isOpen ? 'rotate(180deg)' : 'rotate(0)' }} />
            </button>

            {isOpen && (
              <div style={{ position: 'absolute', width: '100%', marginTop: '0.5rem', backgroundColor: '#1a1a1c', border: '1px solid #303033', borderRadius: '0.75rem', boxShadow: '0 25px 50px -12px rgba(0, 0, 0, 0.5)', zIndex: 50, padding: '0.5rem 0', overflow: 'hidden' }}>
                {languages.map((lang) => (
                  <button
                    key={lang.id}
                    style={{
                      width: '100%',
                      textAlign: 'left',
                      padding: '0.625rem 1.25rem',
                      fontSize: '13px',
                      backgroundColor: language === lang.name ? '#F68B1F' : 'transparent',
                      color: language === lang.name ? 'white' : '#adadb8',
                      border: 'none',
                      cursor: 'pointer'
                    }}
                    onClick={() => {
                      setLanguage(lang.name);
                      setIsOpen(false);
                    }}
                  >
                    {lang.name}
                  </button>
                ))}
              </div>
            )}
          </div>

          {/* Request snippet */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>Example request</h3>
            {renderCode(requestSnippets[language], 'request')}
          </div>

          {/* Response status switcher */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>Example response</h3>
            <div style={{ display: 'flex', flexWrap: 'wrap', gap: '0.5rem', marginBottom: '1rem' }}>
              {statusOrder.map((s) => {
                const active = activeStatus === s;
                return (
                  <button
                    key={s}
                    onClick={() => setActiveStatus(s)}
                    style={{
                      backgroundColor: active ? statusAccent(s) : '#1a1a1c',
                      color: active ? 'white' : '#adadb8',
                      border: `1px solid ${active ? statusAccent(s) : '#303033'}`,
                      padding: '0.4rem 0.75rem',
                      borderRadius: '0.375rem',
                      fontSize: '12px',
                      fontFamily: 'monospace',
                      fontWeight: '600',
                      cursor: 'pointer',
                      transition: 'all 0.15s'
                    }}
                  >
                    {s}
                  </button>
                );
              })}
            </div>
            <div style={{ marginBottom: '0.75rem', fontSize: '13px', color: '#adadb8', lineHeight: '1.5' }}>
              <span style={{ color: statusAccent(activeStatus), fontFamily: 'monospace', fontWeight: '700', marginRight: '0.5rem' }}>{responseExamples[activeStatus].label}</span>
              {responseExamples[activeStatus].desc}
            </div>
            {renderCode(responseExamples[activeStatus].code, `resp-${activeStatus}`)}
          </div>

          {/* Tip Card */}
          <div style={{ backgroundColor: '#1a1a1c', borderRadius: '0.75rem', border: '1px solid #303033', padding: '1.5rem', display: 'flex', flexDirection: 'column', gap: '1rem', borderLeft: '4px solid #F68B1F', boxShadow: '0 20px 25px -5px rgba(0, 0, 0, 0.1)' }}>
            <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
              <h3 style={{ fontSize: '11px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.1em', color: '#88888f', margin: 0 }}>Designing roles</h3>
              <Sparkles size={18} style={{ color: '#F68B1F' }} />
            </div>
            <p style={{ fontSize: '12px', lineHeight: '1.6', color: '#88888f', margin: 0 }}>
              Start narrow. Grant only the privileges a role needs today and add more later — it's much safer than over-provisioning. Pair <code style={{ color: '#e3e3e8', fontFamily: 'monospace' }}>Workspace: Read All Content</code> with task-specific privileges to give read-only auditors a useful view.
            </p>
          </div>

        </div>
      </div>
    </div>
  );
};
