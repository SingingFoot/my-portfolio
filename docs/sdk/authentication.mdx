export const HighlightedCode = ({ code }) => {
  const highlight = (text) => {
    let highlighted = text
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/(\/\/.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/(#.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/("(?:[^"\\]|\\.)*")|('(?:[^'\\]|\\.)*')/g, '<span style="color: #a5d6ff">$1</span>')
      .replace(/\b(import|from|const|let|await|async|new|var|require|return|if|else|public|private|class|interface|package|use|namespace|def|func|true|false|null|try|catch|throw)\b/g, '<span style="color: #ff7b72">$1</span>')
      .replace(/\b(TechWriter|TechWriterClient|TokenManager|RequestOptions|context|ctx|options|client|tw|sc|service)\b/g, '<span style="color: #ffa657">$1</span>')
      .replace(/\b([a-z_][a-z0-9_]*)(?=\s*\()/gi, '<span style="color: #d2a8ff">$1</span>')
      .replace(/\b(\d+)\b/g, '<span style="color: #79c0ff">$1</span>');

    return <code dangerouslySetInnerHTML={{ __html: highlighted }} />;
  };

  return (
    <pre style={{ margin: 0, background: 'transparent', whiteSpace: 'pre-wrap', fontFamily: 'monospace' }}>
      {highlight(code)}
    </pre>
  );
};

              <span>Scope</span>
              <span>Grants</span>
            </div>
            {scopeRows.map((s, idx) => (
              <div key={s.scope} style={bodyRowStyle('220px 1fr', idx === scopeRows.length - 1)}>
                <code style={{ fontFamily: 'monospace', color: '#F68B1F', fontWeight: '600', fontSize: '13px' }}>{s.scope}</code>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{s.what}</span>
              </div>
            ))}
          </div>

          {/* Token lifecycle */}
          <h2 style={{ fontSize: '20px', fontWeight: 'bold', color: 'var(--ifm-font-color-base)', marginBottom: '0.75rem', letterSpacing: '-0.02em' }}>Token lifecycle</h2>
          <p style={{ fontSize: '14px', color: '#adadb8', marginBottom: '1rem' }}>The SDK handles caching, refresh, and retries automatically. The table below describes what happens at each stage so you can reason about edge cases.</p>
          <div style={cardTableStyle}>
            <div style={headerCellsStyle('190px 1fr 1.2fr')}>
              <span>Event</span>
              <span>When it happens</span>
              <span>What to do</span>
            </div>
            {lifecycleRows.map((l, idx) => (
              <div key={l.event} style={bodyRowStyle('190px 1fr 1.2fr', idx === lifecycleRows.length - 1)}>
                <span style={{ color: '#F68B1F', fontWeight: '600' }}>{l.event}</span>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{l.when}</span>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{l.action}</span>
              </div>
            ))}
          </div>

          {/* Authentication errors */}
          <h2 style={{ fontSize: '20px', fontWeight: 'bold', color: 'var(--ifm-font-color-base)', marginBottom: '0.75rem', letterSpacing: '-0.02em' }}>Authentication errors</h2>
          <p style={{ fontSize: '14px', color: '#adadb8', marginBottom: '1rem' }}>Auth failures always include a stable <code style={{ backgroundColor: '#1e1e20', padding: '0.125rem 0.375rem', borderRadius: '0.25rem', fontFamily: 'monospace', color: '#e3e3e8', fontSize: '13px' }}>error.code</code> in the body — branch on the code, not the message.</p>
          <div style={cardTableStyle}>
            <div style={headerCellsStyle('70px 200px 1.2fr 1.4fr')}>
              <span>Status</span>
              <span>error.code</span>
              <span>Likely cause</span>
              <span>How to fix</span>
            </div>
            {errorRows.map((e, idx) => (
              <div key={`${e.code}-${e.err}`} style={bodyRowStyle('70px 200px 1.2fr 1.4fr', idx === errorRows.length - 1)}>
                <span style={{ fontFamily: 'monospace', color: e.code.startsWith('4') ? '#F68B1F' : '#f85149', fontWeight: '700' }}>{e.code}</span>
                <code style={{ fontFamily: 'monospace', color: '#e3e3e8', fontSize: '12px' }}>{e.err}</code>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{e.cause}</span>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{e.fix}</span>
              </div>
            ))}
          </div>

          {/* Best practices */}
          <h2 style={{ fontSize: '20px', fontWeight: 'bold', color: 'var(--ifm-font-color-base)', marginBottom: '0.75rem', letterSpacing: '-0.02em' }}>Security best practices</h2>
          <p style={{ fontSize: '14px', color: '#adadb8', marginBottom: '1rem' }}>The single biggest source of API key incidents is accidental commits. The rules below cost a few minutes each and prevent the painful ones.</p>
          <div style={cardTableStyle}>
            <div style={headerCellsStyle('290px 1fr')}>
              <span>Rule</span>
              <span>Why it matters</span>
            </div>
            {practiceRows.map((p, idx) => (
              <div key={p.rule} style={bodyRowStyle('290px 1fr', idx === practiceRows.length - 1)}>
                <span style={{ color: '#e3e3e8', fontWeight: '600', display: 'flex', alignItems: 'baseline', gap: '0.5rem' }}>
                  <span style={{ color: '#F68B1F' }}>•</span>
                  {p.rule}
                </span>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{p.why}</span>
              </div>
            ))}
          </div>

          {/* Next Steps */}
          <h2 style={{ fontSize: '20px', fontWeight: 'bold', color: 'var(--ifm-font-color-base)', marginBottom: '0.75rem', letterSpacing: '-0.02em' }}>Where to go next</h2>
          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(260px, 1fr))', gap: '1rem', marginBottom: '3rem' }}>
            {nextSteps.map((n) => (
              <a key={n.href} href={n.href} style={{ textDecoration: 'none', backgroundColor: '#1a1a1c', border: '1px solid #303033', borderRadius: '0.625rem', padding: '1.25rem', display: 'flex', flexDirection: 'column', gap: '0.5rem', transition: 'border-color 0.15s' }}>
                <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
                  <span style={{ color: 'var(--ifm-font-color-base)', fontSize: '14px', fontWeight: '700' }}>{n.title}</span>
                  <span style={{ color: '#F68B1F', display: 'flex' }}><ArrowRight size={14} /></span>
                </div>
                <span style={{ color: '#88888f', fontSize: '12px', lineHeight: '1.5' }}>{n.desc}</span>
              </a>
            ))}
          </div>

          {/* Feedback Section */}
          <div style={{ marginTop: '4rem', paddingTop: '3rem', borderTop: '1px solid #1e1e20' }}>
            {!feedback ? (
              <div style={{ display: 'flex', alignItems: 'center', gap: '2rem' }}>
                <span style={{ fontSize: '14px', fontWeight: '500', color: '#88888f' }}>Was this section helpful?</span>
                <button onClick={() => setFeedback('yes')} style={{ color: '#F68B1F', fontSize: '14px', fontWeight: '600', background: 'none', border: 'none', cursor: 'pointer' }}>Yes</button>
                <button onClick={() => setFeedback('no')} style={{ color: '#F68B1F', fontSize: '14px', fontWeight: '600', background: 'none', border: 'none', cursor: 'pointer' }}>No</button>
              </div>
            ) : (
              <div style={{ animation: 'fadeIn 0.5s' }}>
                {feedback === 'yes' ? (
                  <p style={{ fontSize: '14px', color: '#e3e3e8', backgroundColor: '#1a1a1c', padding: '0.75rem 1.5rem', borderRadius: '0.375rem', display: 'inline-block', border: '1px solid #303033' }}>
                    ✨ Thanks for your feedback! We're glad this helped.
                  </p>
                ) : (
                  <div style={{ backgroundColor: '#1a1a1c', padding: '1.5rem', borderRadius: '0.375rem', border: '1px solid #303033', maxWidth: '448px' }}>
                    <p style={{ fontSize: '14px', color: '#e3e3e8', marginBottom: '1rem' }}>We're sorry! How can we improve this page?</p>
                    <textarea
                      style={{ width: '100%', backgroundColor: '#0a0a0a', border: '1px solid #303033', borderRadius: '0.5rem', padding: '0.75rem', fontSize: '0.875rem', color: '#adadb8', outline: 'none' }}
                      placeholder="Your feedback..."
                      rows="4"
                    ></textarea>
                    <button style={{ marginTop: '1rem', backgroundColor: '#F68B1F', color: 'white', fontSize: '0.75rem', fontWeight: 'bold', padding: '0.5rem 1rem', borderRadius: '0.5rem', border: 'none', cursor: 'pointer' }}>Submit Feedback</button>
                  </div>
                )}
              </div>
            )}
            <div style={{ paddingBottom: '8rem' }}></div>
          </div>
        </div>
      </div>

      {/* Right Column: Code */}
      <div style={{ width: isMobileView ? '100%' : '45%', backgroundColor: 'rgba(0,0,0,0.02)', padding: isMobileView ? '1.5rem 1rem' : '3rem', borderLeft: isMobileView ? 'none' : '1px solid #1e1e20', borderTop: isMobileView ? '1px solid #1e1e20' : 'none' }}>
        <div style={{ position: isMobileView ? 'static' : 'sticky', top: isMobileView ? 'auto' : '3rem', display: 'flex', flexDirection: 'column', gap: '2rem' }}>

          {/* Language Picker */}
          <div style={{ position: 'relative' }}>
            <button
              onClick={() => setIsOpen(!isOpen)}
              style={{
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'space-between',
                width: '100%',
                backgroundColor: '#1a1a1c',
                padding: '0.75rem 1.25rem',
                borderRadius: '0.5rem',
                fontSize: '14px',
                color: 'white',
                border: '1px solid #303033',
                cursor: 'pointer'
              }}
            >
              <span style={{ fontWeight: '600' }}>{language}</span>
              <ChevronDown size={16} style={{ color: '#88888f', transform: isOpen ? 'rotate(180deg)' : 'rotate(0)' }} />
            </button>

            {isOpen && (
              <div style={{ position: 'absolute', width: '100%', marginTop: '0.5rem', backgroundColor: '#1a1a1c', border: '1px solid #303033', borderRadius: '0.75rem', boxShadow: '0 25px 50px -12px rgba(0, 0, 0, 0.5)', zIndex: 50, padding: '0.5rem 0', overflow: 'hidden' }}>
                {languages.map((lang) => (
                  <button
                    key={lang.id}
                    style={{
                      width: '100%',
                      textAlign: 'left',
                      padding: '0.625rem 1.25rem',
                      fontSize: '13px',
                      backgroundColor: language === lang.name ? '#F68B1F' : 'transparent',
                      color: language === lang.name ? 'white' : '#adadb8',
                      border: 'none',
                      cursor: 'pointer'
                    }}
                    onClick={() => {
                      setLanguage(lang.name);
                      setIsOpen(false);
                    }}
                  >
                    {lang.name}
                  </button>
                ))}
              </div>
            )}
          </div>

          {/* API key snippet */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>1. Authenticate with an API key</h3>
            {renderCode(apiKeySnippets[language], 'apikey')}
          </div>

          {/* OAuth snippet */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>2. OAuth 2.0 client credentials</h3>
            {renderCode(oauthSnippets[language], 'oauth')}
          </div>

          {/* Refresh snippet */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>3. Refresh tokens</h3>
            {renderCode(refreshSnippets[language], 'refresh')}
          </div>

          {/* Tip Card */}
          <div style={{ backgroundColor: '#1a1a1c', borderRadius: '0.75rem', border: '1px solid #303033', padding: '1.5rem', display: 'flex', flexDirection: 'column', gap: '1rem', borderLeft: '4px solid #F68B1F', boxShadow: '0 20px 25px -5px rgba(0, 0, 0, 0.1)' }}>
            <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
              <h3 style={{ fontSize: '11px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.1em', color: '#88888f', margin: 0 }}>Rotation pattern</h3>
              <Sparkles size={18} style={{ color: '#F68B1F' }} />
            </div>
            <p style={{ fontSize: '12px', lineHeight: '1.6', color: '#88888f', margin: 0 }}>
              Generate the new key first, deploy it alongside the old one, then revoke the old key. The dashboard supports a 24-hour grace period where both keys are valid — long enough to roll a fleet without downtime.
            </p>
          </div>

        </div>
      </div>
    </div>
  );
};
