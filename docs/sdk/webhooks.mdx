export const HighlightedCode = ({ code }) => {
  const highlight = (text) => {
    let highlighted = text
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/(\/\/.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/(#.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/("(?:[^"\\]|\\.)*")|('(?:[^'\\]|\\.)*')/g, '<span style="color: #a5d6ff">$1</span>')
      .replace(/\b(import|from|const|let|await|async|new|var|require|return|if|else|public|private|class|interface|package|use|namespace|def|func|true|false|null|try|catch|throw|when|match|rescue|except)\b/g, '<span style="color: #ff7b72">$1</span>')
      .replace(/\b(TechWriter|TechWriterClient|WebhookVerifier|RequestOptions|context|ctx|options|client|tw|sc|service|err|error|Error|req|res|app|express|hmac|digest)\b/g, '<span style="color: #ffa657">$1</span>')
      .replace(/\b([a-z_][a-z0-9_]*)(?=\s*\()/gi, '<span style="color: #d2a8ff">$1</span>')
      .replace(/\b(\d+)\b/g, '<span style="color: #79c0ff">$1</span>');

    return <code dangerouslySetInnerHTML={{ __html: highlighted }} />;
  };

  return (
    <pre style={{ margin: 0, background: 'transparent', whiteSpace: 'pre-wrap', fontFamily: 'monospace' }}>
      {highlight(code)}
    </pre>
  );
};
