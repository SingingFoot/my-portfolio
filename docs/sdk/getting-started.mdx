export const HighlightedCode = ({ code }) => {
  const highlight = (text) => {
    let highlighted = text
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/(\/\/.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/(#.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/("(?:[^"\\]|\\.)*")|('(?:[^'\\]|\\.)*')/g, '<span style="color: #a5d6ff">$1</span>')
      .replace(/\b(import|from|const|let|await|async|new|var|require|return|if|else|public|private|class|interface|package|use|namespace|def|func|true|false|null|try|catch)\b/g, '<span style="color: #ff7b72">$1</span>')
      .replace(/\b(TechWriter|TechWriterClient|RequestOptions|Article|ArticleParams|context|ctx|options|client|tw|sc|service)\b/g, '<span style="color: #ffa657">$1</span>')
      .replace(/\b([a-z_][a-z0-9_]*)(?=\s*\()/gi, '<span style="color: #d2a8ff">$1</span>')
      .replace(/\b(\d+)\b/g, '<span style="color: #79c0ff">$1</span>');

    return <code dangerouslySetInnerHTML={{ __html: highlighted }} />;
  };

  return (
    <pre style={{ margin: 0, background: 'transparent', whiteSpace: 'pre-wrap', fontFamily: 'monospace' }}>
      {highlight(code)}
    </pre>
  );
};

                <span style={{ fontSize: '11px', fontWeight: '700', textTransform: 'uppercase', letterSpacing: '0.05em', color: c.required ? '#f85149' : '#6b6b7b' }}>{c.required ? 'required' : 'optional'}</span>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{c.desc}</span>
              </div>
            ))}
          </div>

          {/* Core concepts */}
          <h2 style={{ fontSize: '20px', fontWeight: 'bold', color: 'var(--ifm-font-color-base)', marginBottom: '0.75rem', letterSpacing: '-0.02em' }}>Core concepts</h2>
          <p style={{ fontSize: '14px', color: '#adadb8', marginBottom: '1rem' }}>A short tour of the domain model. Every SDK call maps onto one of these resources.</p>
          <div style={cardTableStyle}>
            <div style={headerCellsStyle('170px 1fr')}>
              <span>Concept</span>
              <span>Description</span>
            </div>
            {conceptRows.map((c, idx) => (
              <div key={c.c} style={bodyRowStyle('170px 1fr', idx === conceptRows.length - 1)}>
                <span style={{ color: '#F68B1F', fontWeight: '600' }}>{c.c}</span>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{c.d}</span>
              </div>
            ))}
          </div>

          {/* Features */}
          <h2 style={{ fontSize: '20px', fontWeight: 'bold', color: 'var(--ifm-font-color-base)', marginBottom: '0.75rem', letterSpacing: '-0.02em' }}>What ships in the box</h2>
          <p style={{ fontSize: '14px', color: '#adadb8', marginBottom: '1rem' }}>You don't have to glue these features together yourself — they're built into every language binding.</p>
          <div style={cardTableStyle}>
            <div style={headerCellsStyle('230px 1fr')}>
              <span>Feature</span>
              <span>What it does</span>
            </div>
            {featureRows.map((f, idx) => (
              <div key={f.f} style={bodyRowStyle('230px 1fr', idx === featureRows.length - 1)}>
                <span style={{ color: '#e3e3e8', fontWeight: '600' }}>{f.f}</span>
                <span style={{ color: '#adadb8', lineHeight: '1.5' }}>{f.d}</span>
              </div>
            ))}
          </div>

          {/* Next Steps */}
          <h2 style={{ fontSize: '20px', fontWeight: 'bold', color: 'var(--ifm-font-color-base)', marginBottom: '0.75rem', letterSpacing: '-0.02em' }}>Where to go next</h2>
          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(260px, 1fr))', gap: '1rem', marginBottom: '3rem' }}>
            {nextSteps.map((n) => (
              <a key={n.href} href={n.href} style={{ textDecoration: 'none', backgroundColor: '#1a1a1c', border: '1px solid #303033', borderRadius: '0.625rem', padding: '1.25rem', display: 'flex', flexDirection: 'column', gap: '0.5rem', transition: 'border-color 0.15s' }}>
                <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
                  <span style={{ color: 'var(--ifm-font-color-base)', fontSize: '14px', fontWeight: '700' }}>{n.title}</span>
                  <span style={{ color: '#F68B1F', display: 'flex' }}><ArrowRight size={14} /></span>
                </div>
                <span style={{ color: '#88888f', fontSize: '12px', lineHeight: '1.5' }}>{n.desc}</span>
              </a>
            ))}
          </div>

          {/* Feedback Section */}
          <div style={{ marginTop: '4rem', paddingTop: '3rem', borderTop: '1px solid #1e1e20' }}>
            {!feedback ? (
              <div style={{ display: 'flex', alignItems: 'center', gap: '2rem' }}>
                <span style={{ fontSize: '14px', fontWeight: '500', color: '#88888f' }}>Was this section helpful?</span>
                <button onClick={() => setFeedback('yes')} style={{ color: '#F68B1F', fontSize: '14px', fontWeight: '600', background: 'none', border: 'none', cursor: 'pointer' }}>Yes</button>
                <button onClick={() => setFeedback('no')} style={{ color: '#F68B1F', fontSize: '14px', fontWeight: '600', background: 'none', border: 'none', cursor: 'pointer' }}>No</button>
              </div>
            ) : (
              <div style={{ animation: 'fadeIn 0.5s' }}>
                {feedback === 'yes' ? (
                  <p style={{ fontSize: '14px', color: '#e3e3e8', backgroundColor: '#1a1a1c', padding: '0.75rem 1.5rem', borderRadius: '0.375rem', display: 'inline-block', border: '1px solid #303033' }}>
                    ✨ Thanks for your feedback! We're glad this helped.
                  </p>
                ) : (
                  <div style={{ backgroundColor: '#1a1a1c', padding: '1.5rem', borderRadius: '0.375rem', border: '1px solid #303033', maxWidth: '448px' }}>
                    <p style={{ fontSize: '14px', color: '#e3e3e8', marginBottom: '1rem' }}>We're sorry! How can we improve this page?</p>
                    <textarea
                      style={{ width: '100%', backgroundColor: '#0a0a0a', border: '1px solid #303033', borderRadius: '0.5rem', padding: '0.75rem', fontSize: '0.875rem', color: '#adadb8', outline: 'none' }}
                      placeholder="Your feedback..."
                      rows="4"
                    ></textarea>
                    <button style={{ marginTop: '1rem', backgroundColor: '#F68B1F', color: 'white', fontSize: '0.75rem', fontWeight: 'bold', padding: '0.5rem 1rem', borderRadius: '0.5rem', border: 'none', cursor: 'pointer' }}>Submit Feedback</button>
                  </div>
                )}
              </div>
            )}
            <div style={{ paddingBottom: '8rem' }}></div>
          </div>
        </div>
      </div>

      {/* Right Column: Install + Initialize + Example */}
      <div style={{ width: isMobileView ? '100%' : '45%', backgroundColor: 'rgba(0,0,0,0.02)', padding: isMobileView ? '1.5rem 1rem' : '3rem', borderLeft: isMobileView ? 'none' : '1px solid #1e1e20', borderTop: isMobileView ? '1px solid #1e1e20' : 'none' }}>
        <div style={{ position: isMobileView ? 'static' : 'sticky', top: isMobileView ? 'auto' : '3rem', display: 'flex', flexDirection: 'column', gap: '2rem' }}>

          {/* Package Manager Picker */}
          <div style={{ position: 'relative' }}>
            <button
              onClick={() => setPmOpen(!pmOpen)}
              style={{
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'space-between',
                width: '100%',
                backgroundColor: '#1a1a1c',
                padding: '0.75rem 1.25rem',
                borderRadius: '0.5rem',
                fontSize: '14px',
                color: 'white',
                border: '1px solid #303033',
                cursor: 'pointer'
              }}
            >
              <span>
                <span style={{ fontWeight: '600' }}>{packageManagers.find((p) => p.id === pm).name}</span>
                <span style={{ color: '#88888f', marginLeft: '0.5rem', fontSize: '12px' }}>· {packageManagers.find((p) => p.id === pm).lang}</span>
              </span>
              <ChevronDown size={16} style={{ color: '#88888f', transform: pmOpen ? 'rotate(180deg)' : 'rotate(0)' }} />
            </button>

            {pmOpen && (
              <div style={{ position: 'absolute', width: '100%', marginTop: '0.5rem', backgroundColor: '#1a1a1c', border: '1px solid #303033', borderRadius: '0.75rem', boxShadow: '0 25px 50px -12px rgba(0, 0, 0, 0.5)', zIndex: 50, padding: '0.5rem 0', overflow: 'hidden' }}>
                {packageManagers.map((p) => (
                  <button
                    key={p.id}
                    style={{
                      width: '100%',
                      textAlign: 'left',
                      padding: '0.625rem 1.25rem',
                      fontSize: '13px',
                      backgroundColor: pm === p.id ? '#F68B1F' : 'transparent',
                      color: pm === p.id ? 'white' : '#adadb8',
                      border: 'none',
                      cursor: 'pointer',
                      display: 'flex',
                      justifyContent: 'space-between',
                      alignItems: 'center'
                    }}
                    onClick={() => {
                      setPm(p.id);
                      setPmOpen(false);
                    }}
                  >
                    <span>{p.name}</span>
                    <span style={{ color: pm === p.id ? 'rgba(255,255,255,0.7)' : '#6b6b7b', fontSize: '11px' }}>{p.lang}</span>
                  </button>
                ))}
              </div>
            )}
          </div>

          {/* Install snippet */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>1. Install</h3>
            {renderCode(installSnippets[pm], 'install')}
          </div>

          {/* Init snippet */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>2. Initialize the client</h3>
            {renderCode(initSnippets[pm], 'init')}
          </div>

          {/* Example snippet */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>3. Make your first call</h3>
            {renderCode(exampleSnippets[pm], 'example')}
          </div>

          {/* Tip Card */}
          <div style={{ backgroundColor: '#1a1a1c', borderRadius: '0.75rem', border: '1px solid #303033', padding: '1.5rem', display: 'flex', flexDirection: 'column', gap: '1rem', borderLeft: '4px solid #F68B1F', boxShadow: '0 20px 25px -5px rgba(0, 0, 0, 0.1)' }}>
            <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
              <h3 style={{ fontSize: '11px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.1em', color: '#88888f', margin: 0 }}>Pro tip</h3>
              <Sparkles size={18} style={{ color: '#F68B1F' }} />
            </div>
            <p style={{ fontSize: '12px', lineHeight: '1.6', color: '#88888f', margin: 0 }}>
              Read your API key from an environment variable and load it once at startup. Hard-coding the key — even in dev — is the most common cause of accidental exposure on public repos.
            </p>
          </div>

        </div>
      </div>
    </div>
  );
};
