export const HighlightedCode = ({ code }) => {
  const highlight = (text) => {
    let highlighted = text
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/(\/\/.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/(#.*$)/gm, '<span style="color: #8b949e">$1</span>')
      .replace(/("(?:[^"\\]|\\.)*")|('(?:[^'\\]|\\.)*')/g, '<span style="color: #a5d6ff">$1</span>')
      .replace(/\b(import|from|const|let|await|async|new|var|require|return|if|else|public|private|class|interface|package|use|namespace|def|func|true|false|null|try|catch|throw|while|for|switch|case|break|continue|in|of)\b/g, '<span style="color: #ff7b72">$1</span>')
      .replace(/\b(TechWriter|TechWriterClient|TechWriterError|RateLimitError|RequestOptions|Article|ArticleParams|context|ctx|options|client|tw|sc|service|err|error|Error|randomUUID)\b/g, '<span style="color: #ffa657">$1</span>')
      .replace(/\b([a-z_][a-z0-9_]*)(?=\s*\()/gi, '<span style="color: #d2a8ff">$1</span>')
      .replace(/\b(\d+)\b/g, '<span style="color: #79c0ff">$1</span>');

    return <code dangerouslySetInnerHTML={{ __html: highlighted }} />;
  };

  return (
    <pre style={{ margin: 0, background: 'transparent', whiteSpace: 'pre-wrap', fontFamily: 'monospace' }}>
      {highlight(code)}
    </pre>
  );
};

              <div style={{ position: 'absolute', width: '100%', marginTop: '0.5rem', backgroundColor: '#1a1a1c', border: '1px solid #303033', borderRadius: '0.75rem', boxShadow: '0 25px 50px -12px rgba(0, 0, 0, 0.5)', zIndex: 50, padding: '0.5rem 0', overflow: 'hidden', maxHeight: '420px', overflowY: 'auto' }}>
                {scenarios.map((s) => (
                  <button
                    key={s.id}
                    style={{
                      width: '100%',
                      textAlign: 'left',
                      padding: '0.625rem 1.25rem',
                      fontSize: '13px',
                      backgroundColor: scenarioId === s.id ? '#F68B1F' : 'transparent',
                      color: scenarioId === s.id ? 'white' : '#adadb8',
                      border: 'none',
                      cursor: 'pointer',
                      display: 'flex',
                      justifyContent: 'space-between',
                      alignItems: 'center'
                    }}
                    onClick={() => {
                      setScenarioId(s.id);
                      setScenarioOpen(false);
                    }}
                  >
                    <span>{s.name}</span>
                    <span style={{ color: scenarioId === s.id ? 'rgba(255,255,255,0.7)' : '#6b6b7b', fontSize: '11px' }}>{s.desc}</span>
                  </button>
                ))}
              </div>
            )}
          </div>

          {/* Code snippet */}
          <div>
            <h3 style={{ color: 'var(--ifm-font-color-base)', fontSize: '13px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.08em', marginBottom: '0.875rem' }}>Pattern — {selected.name}</h3>
            {renderCode(codeSnippets[scenarioId], 'snippet-' + scenarioId)}
          </div>

          {/* Tip Card */}
          <div style={{ backgroundColor: '#1a1a1c', borderRadius: '0.75rem', border: '1px solid #303033', padding: '1.5rem', display: 'flex', flexDirection: 'column', gap: '1rem', borderLeft: '4px solid #F68B1F', boxShadow: '0 20px 25px -5px rgba(0, 0, 0, 0.1)' }}>
            <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
              <h3 style={{ fontSize: '11px', fontWeight: 'bold', textTransform: 'uppercase', letterSpacing: '0.1em', color: '#88888f', margin: 0 }}>Pro tip</h3>
              <Sparkles size={18} style={{ color: '#F68B1F' }} />
            </div>
            <p style={{ fontSize: '12px', lineHeight: '1.6', color: '#88888f', margin: 0 }}>
              Caching doesn't just save money; it prevents <span style={{ color: '#F68B1F' }}>thundering herd</span> problems in your own infrastructure. If multiple nodes start at once, local ETags ensure they don't all pull the same 5MB payload from TechWriter simultaneously.
            </p>
          </div>

        </div>
      </div>
    </div>
  );
};
