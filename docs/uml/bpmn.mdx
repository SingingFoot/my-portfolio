```mermaid
graph LR
    %% Direction left to right
    direction LR

    subgraph Client_Lane [ ]
        direction LR
        L1[<b>CLIENT</b>] --- Start((Start))
        Start --> Insert[Insert money]
        Insert --> Select[Select drink]
        Select --> Grab[Grab drink]
        Grab --> Finish((Finish))
    end

    subgraph VM_Lane [ ]
        direction LR
        L2[<b>VENDING MACHINE</b>] --- Check[Check money]
        Check --- Stock{In stock?}
        Stock -- No --> Out[Out of stock]
        Stock -- Yes --> Dispense[Dispense drink]
        Dispense --> Wait((Wait 3s))

        %% Placeholder node to balance lane length
        Wait ~~~ Placeholder[ ]
    end

    %% Cross-lane flows
    Insert --> Check
    Check --> Select
    Select --> Stock
    Out --> Select
    Wait --> Grab

    %% Styling
    style L1 fill:none,stroke:none
    style L2 fill:none,stroke:none
    style Client_Lane fill:none,stroke:#666,stroke-dasharray: 5 5
    style VM_Lane fill:none,stroke:#666,stroke-dasharray: 5 5
    style Out fill:none,stroke:#ff6666,stroke-width:2px
    style Dispense fill:none,stroke:#33cc33,stroke-width:2px
    style Placeholder fill:none,stroke:none
```
